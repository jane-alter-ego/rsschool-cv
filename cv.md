  // character count in string case insensitive
  const countChars = (str, char) => {
    let i = 0;
    let count = 0;
    while (i < str.length) {
      if (str[i] === char.toLowerCase() || str[i] === char.toUpper.Case()) {
        count++;
      }
      i++;
    }
    return count;
  };
